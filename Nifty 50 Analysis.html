import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

file_name = 'Nifty 50.xlsx'
df = pd.read_excel(file_name)
df['Date'] = pd.to_datetime(df['Date'])

stocks = df['Ticker'].unique() 
returns = []

for stock in stocks:
    stock_data = df[df['Ticker'] == stock].sort_values('Date')
    
    first_price = stock_data['Adj Close'].iloc[0]
    last_price = stock_data['Adj Close'].iloc[-1]
    
    start_date = stock_data['Date'].iloc[0]
    end_date = stock_data['Date'].iloc[-1]
    years = (end_date - start_date).days / 365.25
    
    if years > 0:
        annualized_return = ((last_price / first_price) ** (1 / years) - 1) * 100
    else:
        annualized_return = 0
        
    returns.append({
        'Ticker': stock, 
        'Annualized_Return_%': annualized_return,
        'Years_Held': round(years, 1),
        'Start_Date': start_date.strftime('%Y-%m-%d'),
        'End_Date': end_date.strftime('%Y-%m-%d')
    })

returns_df = pd.DataFrame(returns).sort_values(by='Annualized_Return_%', ascending=False)

print("Period Summary")
print(f"Start: {returns_df['Start_Date'].min()}")
print(f"End:   {returns_df['End_Date'].max()}")
print(f"Average Stock Return: {returns_df['Annualized_Return_%'].mean():.2f}%")

print("Top 5 Performers (Annualised)")
print(returns_df[['Ticker', 'Annualized_Return_%', 'Years_Held']].head(5).to_string(index=False))

print("Bottom 5 Performers (Annualised)")
print(returns_df[['Ticker', 'Annualized_Return_%', 'Years_Held']].tail(5).to_string(index=False))

plt.figure(figsize=(10, 6))
plt.hist(returns_df['Annualized_Return_%'], bins=15, color='skyblue', edgecolor='black')
plt.axvline(returns_df['Annualized_Return_%'].mean(), color='red', linestyle='dashed', linewidth=2, 
            label=f"Avg Stock: {returns_df['Annualized_Return_%'].mean():.2f}%")

plt.title('Distribution of Annualized Returns (Nifty 50 Stocks)', fontsize=14)
plt.xlabel('Annualised Return (CAGR %)', fontsize=12)
plt.ylabel('Number of Stocks', fontsize=12)
plt.legend()
plt.grid(axis='y', alpha=0.3)

plt.show()

from nselib import capital_market
import pandas as pd

start_date = '01-01-2006'
end_date = '25-05-2024'

data = capital_market.index_data(index='NIFTY 50', from_date=start_date, to_date=end_date)

data['TIMESTAMP'] = pd.to_datetime(data['TIMESTAMP'], format='%d-%b-%Y')
data = data.sort_values('TIMESTAMP')

actual_start_date = data['TIMESTAMP'].iloc[0].strftime('%d-%b-%Y')
actual_end_date = data['TIMESTAMP'].iloc[-1].strftime('%d-%b-%Y')

start_val = float(str(data.iloc[0]['CLOSE_INDEX_VAL']).replace(',', ''))
end_val = float(str(data.iloc[-1]['CLOSE_INDEX_VAL']).replace(',', ''))

days_diff = (data['TIMESTAMP'].iloc[-1] - data['TIMESTAMP'].iloc[0]).days
years = days_diff / 365.25

cagr = (((end_val / start_val) ** (1 / years)) - 1) * 100

print(f"Index Performance Analysis")
print(f"Period:       {actual_start_date} to {actual_end_date}")
print(f"Total Days:   {days_diff}")
print(f"Total Years:  {years:.2f}")
print(f"Start Value:  {start_val}")
print(f"End Value:    {end_val}")
print(f"Annualised Return (CAGR): {cagr:.2f}%")

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt

df = pd.read_excel('Nifty 50.xlsx')
df['Date'] = pd.to_datetime(df['Date'])

# Reshape data to have tickers as columns
pivot_df = df.pivot(index='Date', columns='Ticker', values='Adj Close')

# Calculate daily percentage returns
returns_df = pivot_df.pct_change().dropna()

# Create a professional Benchmark (Equal-weighted average of all constituents)
returns_df['Market_Benchmark'] = returns_df.mean(axis=1)

# Calculate correlations against the Benchmark
correlations = returns_df.corr()['Market_Benchmark'].drop('Market_Benchmark').sort_values(ascending=False)

print("\n--- 1. High Correlation (Market Proxies) ---")
print("These stocks have the highest Beta-like relationship with the index.")
print(correlations.head(5).to_string())

print("\n--- 2. Low Correlation (Diversifiers) ---")
print("These stocks offer idiosyncratic returns, ideal for risk diversification.")
print(correlations.tail(5).to_string())

top_mirror = correlations.index[0]
top_outlier = correlations.index[-1]

# Visualization
fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(16, 6))

# High correlation plot
sns.regplot(x=returns_df[top_mirror], y=returns_df['Market_Benchmark'], ax=ax1, 
            scatter_kws={'alpha':0.3}, line_kws={'color':'red'})
ax1.set_title(f'{top_mirror} vs Market Benchmark\n(Strong Positive Correlation)')

# Low correlation plot

            scatter_kws={'alpha':0.3}, line_kws={'color':'blue'})
ax2.set_title(f'{top_outlier} vs Market Benchmark\n(Low Correlation / Diversifier)')

plt.tight_layout()
plt.show()

# Correlation Heatmap
plt.figure(figsize=(14, 10))
sns.heatmap(returns_df.drop(columns=['Market_Benchmark']).corr(), 
            cmap='coolwarm', center=0, annot=False)
plt.title('Nifty 50: Constituent Correlation Matrix', fontsize=15)
plt.show()

Pt 4. Performs a Risk-Adjusted Performance Analysis, calculating Sharpe Ratio, Beta, and Max Drawdown, identifying stocks with best risk-to-reward profiles and market resilience.

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns

df = pd.read_excel('Nifty 50.xlsx')
df['Date'] = pd.to_datetime(df['Date'])
pivot_df = df.pivot(index='Date', columns='Ticker', values='Adj Close').sort_index()

returns_df = pivot_df.pct_change().dropna()

market_returns = returns_df.mean(axis=1)

metrics = []
risk_free_rate = 0.06  # Assuming 6% annual risk-free rate for India

for stock in returns_df.columns:
    days = (pivot_df[stock].dropna().index[-1] - pivot_df[stock].dropna().index[0]).days
    total_return = (pivot_df[stock].iloc[-1] / pivot_df[stock].iloc[0])
    ann_return = (total_return ** (365.25 / days)) - 1
    
    ann_vol = returns_df[stock].std() * np.sqrt(252)
    
    sharpe = (ann_return - risk_free_rate) / ann_vol
    
    covariance = returns_df[stock].cov(market_returns)
    market_variance = market_returns.var()
    beta = covariance / market_variance
    
    cumulative = (1 + returns_df[stock]).cumprod()
    peak = cumulative.cummax()
    drawdown = (cumulative - peak) / peak
    max_drawdown = drawdown.min()
    
    metrics.append({
        'Ticker': stock,
        'Ann_Return': ann_return * 100,
        'Ann_Vol': ann_vol * 100,
        'Sharpe_Ratio': sharpe,
        'Beta': beta,
        'Max_Drawdown_%': max_drawdown * 100
    })

analysis_df = pd.DataFrame(metrics).sort_values('Sharpe_Ratio', ascending=False)


plt.figure(figsize=(12, 7))
sns.scatterplot(data=analysis_df, x='Ann_Vol', y='Ann_Return', size='Sharpe_Ratio', hue='Beta', palette='viridis', sizes=(50, 400))
plt.title('Nifty 50: Volatility vs. Return', fontsize=15)
plt.xlabel('Annualised Volatility (%)', fontsize=12)
plt.ylabel('Annualised Return (CAGR %)', fontsize=12)
plt.grid(True, alpha=0.3)

for i in range(5):
    plt.annotate(analysis_df.iloc[i]['Ticker'], (analysis_df.iloc[i]['Ann_Vol'], analysis_df.iloc[i]['Ann_Return']))

plt.show()

# --- PRINT SUMMARY ---
